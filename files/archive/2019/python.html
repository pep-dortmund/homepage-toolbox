def primes(max_num):
    # Only primes smaller than max_num are calculated
    is_prime = max_num * [True]
    is_prime[0] = False
    is_prime[1] = False

    primes = []

    # Sieve of Erathosthenes:
    for i in range(2, max_num):
        if is_prime[i]:
            for j in range(2 * i, max_num, i):
                # Multiples are not primes
                is_prime[j] = False
            primes.append(i)
    
    return primes

print(primes(100))

gibt die Fehlermeldung einen Traceback aus.
Dieser Fall ist eher die Regel, vorallem, wenn Module verwendet werden.
Der Traceback zeigt die Reihenfolge aller Funktionsaufrufe, die
am Ende zu dem Fehler gefhrt haben mit dem letzten Aufruf ganz unten (most recent call last).

Debugging leicht gemacht

Es kann vorkommen, dass man mal berhaupt nicht wei, warum etwas nicht funktioniert.

In diesem Fall sollte man zunchst mal Ruhe bewahren
und sich dann auf systematische Fehlersuche begeben.
Ein einfacher Ansatz ist das print()-debugging.

Kleiner Trick, um Schuhgre und Alter einer weiteren Person zu berechnen. Die zweite Person muss folgende Anweisungen befolgen:

1. Multipliziere deine Schuhgre mit 5         
2. Addiere 50 zum Ergebnis                      
3. Multipliziere das Ergebnis mit 20            
4. Addiere 1018 zum Ergebnis                    
5. Ziehe dein Geburtsjahr vom Ergebnis ab
6. Nenne das Ergebnis
In[100]:
def shoesize_and_age(shoesize, birth_year):
    step_1 = shoesize * 5
    step_2 = step_1 + 50
    step_3 = step_2 * 20
    step_4 = step_3 + 1018
    step_5 = step_4 - birth_year
 
    return step_5
